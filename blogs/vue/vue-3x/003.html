003 vux3.x之动画和过渡

JsQQ 2021-06-23 Vue API

# 概念

  • 动画
    一个DOM元素，从一个地方移到另外一个地方。
  • 过渡
    是DOM元素中的一个属性，缓慢的变成为另外一个属性。

# 按钮控制动画

<style>
  .animation {
    animation: leftToRight 3s;
  }

  @keyframes leftToRight {
    0% {
      transform: translateX(0px);
    }
    25% {
      transform: translateX(250px);
    }
    50% {
      transform: translateX(500px);
    }
    75% {
      transform: translateX(750px);
    }
    100% {
      transform: translateX(1000px);
    }
  }
</style>

<script>
  const app = Vue.createApp({
    data(){
      return {
        isAnimate: {
          animation: false
        }
      }
    },
    methods: {
      handleClick(){
        this.isAnimate.animation = !this.isAnimate.animation
      }
    },
    template:`
      <div :class='isAnimate'>
        <button @click='handleClick'>启动/关闭</button>
      </div>
    `
  })

  const vm = app.mount('#app')
</script>

# 按钮控制过渡

<style>
  .transition {
    transition: 3s background-color ease;
  }

  .red {
    background-color: red;
  }

  .yellow {
    background-color: yellow;
  }
</style>

<script>
  const app = Vue.createApp({
    data(){
      return {
        css:{
          transition: true,
          red: true,
          yellow: false

    methods: {
      handleClick(){

    template:`
      <div :class='css'>blog</div>
      <button @click="handleClick">切换颜色</button>
    `
  })
  const vm = app.mount('#app')
</script>

# vue3制作动画

  • <transition></transition>包裹的元素就支持动画
  • .v-enter-active{} 进入动画
  • .v-leave-active{} 离开动画
<style>
  @keyframes comein{
    0%{
      transform:translateX(-100px)
    }
    100%{
      transform:translateX(50px)
    }
  }
  @keyframes comeout{
    0%{
      transform:translateX(50px)
    }
    100%{
      transform:translateX(-100px)
    }
  }
  .v-enter-active{
    animation: comein 1s;
  }
  .v-leave-active{
    animation : comeout 1s;
  }
</style>

<script>
  const app = Vue.createApp({
    data(){
      return {
        isShow: false
      }
    },
    methods: {
      handleClick(){
        this.isShow = !this.isShow
      }
    },
    template:`
      <transition>
        <div v-if="isShow">blog</div>
      </transition>
      <div>
        <button @click='handleClick'>切换动画</button>  
      </div>
    `
  })
  const vm = app.mount('#app')
</script>

# transition的技巧

  • type='animation':不管过渡时长是多少，动画结束，整个过程结束
  • :duration="1000": 1秒后，结束动画和过渡。
  • :duration="{enter:1000,leave:3000}": 入场的时候1秒种，出厂的时候3秒钟
  • :css="false": 不使用动画
  • @before-enter="fn"：在动画开始前执行的函数
    handleBeforeEnter(element){
  element.style.color = 'red'
},

# 动态组件实现动画切换

<style>
  .v-enter-from,
  .v-leave-to {
    opacity: 0;
  }
  .v-enter-to,
  .v-leave-from {
    opacity: 1;
  }
  .v-enter-active,
  .v-leave-active {
    transition: opacity 2s ease-in;
  }
</style>

<script>
  const f1 = {
    template:`<div>blog F1 ...</div>`
  }

  const f2 = {
    template:`<div>blog F2 ...</div>`
  }

  const app = Vue.createApp({
    data(){
      return {
        isShow: false,
        component: 'f1',
      }
    },
    methods: {
      handleClick(){
        this.component = this.component === 'f1' ? 'f2': 'f1'
      }
    },
    template:`
      <transition
        mode="out-in"
        appear
      >
        <component :is='component'/>  // 动态组件
      </transition>
      <div>
        <button @click='handleClick'>切换动画</button>  
      </div>
    `
  })
  const vm = app.mount('#app')
</script>
