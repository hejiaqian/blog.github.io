# 组件间的传值

# 父组件采用:动态绑定值，子组件采用props接收值

const app = Vue.createApp({
  data(){
    return {
      name:'父组件给子组件携带的name值',
      age:22,
      pay:()=>{
        alert('给你500元')
      }
    }
  },
  template:`<h2>组件</h2>
            <Child :name='name' :age='age' :pay='pay'/>
  `
  // 注意：动态绑定值可以是多种类型；静态参数只是string类型。
})

app.component('Child',{
  props:['name','age','pay'],
  template:`<div>{{name}} - {{ typeof age}}<button @click='this.pay'>触发父组件的事件</button></div>`
})
const vm = app.mount('#app')

# 传递值的校验

  • 类型: type: String、Boolean、Array、Object、Function、Symbol
  • 必填值: required: true / false
  • 默认值: default:'默认值'
  • 精准校验: validator:value => { return value.search('校验的结果') != -1 }
props:{
  name: {
    type: String,
    required: true,
    default:'默认值',
    validator: value => {
      return value.search('校验的结果') != -1 // 判定如果不为-1就是通过验证
    }
  }
}

# 当项数据流

  • 数据从父级组件传递给子组件，只能单向绑定。子组件内部不能直接修改从父组件传递过来的数据。

# Non-Props属性

  • 父组件向子组件传递参数的时候，子组件不写任何的接受方法。
  • 父组件就会直接把属性完全复制给子组件，子组件也是可以得到属性值的。
const app = Vue.createApp({
  data(){
    return {
      counter: 0
    } 
  },
  template:`
    <h2>组件</h2>
    <Hello msg='no-props' style="color:red;"/>
  `
})

// 全局组件
app.component('Hello',{
  // props:['msg'], // 指定参数属性
  // inheritAttrs: false, // 不接受参数属性
  mounted() {
    console.log(this.$attrs) // 常用属性
  },
  template:`
    <h3 v-bind='$attrs'>Hello Word</h3> <!-- 父组件传递的所以参数属性都复制到子组件当中 -->
    <h3 v-bind:style='$attrs.style'>Hello Word</h3> <!-- 指定style属性 -->
    <h3>Hello Word</h3>
  `
})

const vm = app.mount('#app')

# 父子组件通信

  • 父组件通过@事件名='事件方法'
  • 子组件通过emits注册事件名，采用this.$emit(事件名)调用
const app = Vue.createApp({
  data(){
    return {
      counter: 0
    } 
  },
  methods: {
    handleAddCounter(num){
      // this.counter += num
      this.counter = num
    }
  },
  template:`
    <h2>组件</h2>
    <counter :counter='counter' @add='handleAddCounter'/>  
  `
})

app.component('Counter',{
  props:['counter'],
  // emits:['add'],
  emits:{
    add: value => {
      return value < 20 ? true : false  // 超过20就提示：Invalid event arguments: event validation failed for event "add".
    }
  },
  methods:{
    handleClick(){
      // this.$emit('add',2)
      this.$emit('add', this.counter + 3)
    }
  },
  template:`
    {{counter}}<button @click='handleClick'>增加数量</button>
  `
})

const vm = app.mount('#app')

# 基本使用：

<slot></slot>
1
  • 可以插入dom，文字，组件
    注意：
    • 父模板里调用的数据属性，使用的都是父模板里的数据。
    • 子模板里调用的数据属性，使用的都是子模板里的数据。

<slot>

</slot>

# 作用域插槽

const app = Vue.createApp({
  template:`
    <h2>欢迎：</h2>
    <!-- <list /> -->
    <list v-slot='props'>
      <div>{{props.item}}--{{props}}</div>
    </list>

    <!-- es6写法 -->
    <list v-slot="{item, ...props}">
      <div>{{item}}--{{props}}</div>  
    </list>
  `
})

app.component('jsQ',{
  template:`
    <div>你选择了
    </div>
  `
})

app.component('List',{
  data(){
    return {
      list: ['测试1','测试2','测试3','测试4']
    }
  },
  template:`
    <div>
      <!-- <div v-for="(item,index) in list" :key='index'>{{item}}</div>  -->
      <slot v-for="(item,index) in list" :key='index' :item='item'/>
    </div>
  `
})

const vm = app.mount('#app')
